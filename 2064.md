class Solution {
public:
    int minimizedMaximum(int n, vector<int>& quantities) {
        int maxi=0;
        for(int q : quantities){
            if(maxi < q){
                maxi=q;
            }
        }
        int left = 1, right = maxi;
        
        while (left < right) {
            int mid = left + (right - left) / 2;
            int stores = 0;
            
            for (int q : quantities) {
                stores += (q + mid - 1) / mid; //ceil((double)q / mid);
            }
            
            if (stores > n) {
                left = mid + 1;
            } else {
                right = mid;
            }
        }
        
        return left;
    }
};
