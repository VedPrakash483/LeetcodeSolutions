class Solution {
public:
    int getLucky(string s, int k) {
        string digits = getnum(s);
        int result = 0;
        for (char c : digits) {
            result += c - '0';
        }

        for (int i = 1; i < k; i++) {
            result = kloop(result);
        }

        return result;
    }

    string getnum(string str) {
        string digits = "";
        for (char c : str) {
            int num = c - 'a' + 1;
            digits += to_string(num);
        }
        return digits;
    }

    int kloop(int number) {
        int sum = 0;
        while (number != 0) {
            sum += number % 10;
            number /= 10;
        }
        return sum;
    }
};
